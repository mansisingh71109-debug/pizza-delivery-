// Select DOM elements
const orderPizzaBtn = document.getElementById("orderPizzaBtn");
const trackingStatus = document.getElementById("trackingStatus");

// Pizza customization options and prices
const pizzaOptions = {
  size: {
    small: 8,
    medium: 10,
    large: 12,
  },
  crust: {
    thin: 2,
    thick: 3,
    stuffed: 4,
  },
  toppings: {
    pepperoni: 1,
    mushrooms: 1,
    onions: 1,
    olives: 1,
    cheese: 1,
  },
};

// Function to customize pizza order
function customizePizzaOrder() {
  // Prompt user for size
  const size = prompt(
    "Choose your pizza size: small ($8), medium ($10), large ($12)"
  ).toLowerCase();

  if (!pizzaOptions.size[size]) {
    alert("Invalid size selected. Please try again.");
    return;
  }

  // Prompt user for crust
  const crust = prompt(
    "Choose your crust: thin ($2), thick ($3), stuffed ($4)"
  ).toLowerCase();

  if (!pizzaOptions.crust[crust]) {
    alert("Invalid crust selected. Please try again.");
    return;
  }

  // Prompt user for toppings
  const toppings = prompt(
    "Choose your toppings (comma-separated): pepperoni, mushrooms, onions, olives, cheese ($1 each)"
  )
    .toLowerCase()
    .split(",")
    .map((topping) => topping.trim());

  // Validate toppings
  const invalidToppings = toppings.filter(
    (topping) => !pizzaOptions.toppings[topping]
  );

  if (invalidToppings.length > 0) {
    alert(`Invalid toppings selected: ${invalidToppings.join(", ")}. Please try again.`);
    return;
  }

  // Calculate total price
  const sizePrice = pizzaOptions.size[size];
  const crustPrice = pizzaOptions.crust[crust];
  const toppingsPrice = toppings.length * pizzaOptions.toppings.pepperoni; // All toppings cost $1
  const totalPrice = sizePrice + crustPrice + toppingsPrice;

  // Display order summary
  trackingStatus.innerHTML = `
    <strong>Order Summary:</strong><br>
    Size: ${size} ($${sizePrice})<br>
    Crust: ${crust} ($${crustPrice})<br>
    Toppings: ${toppings.join(", ")} ($${toppingsPrice})<br>
    <strong>Total Price:</strong> $${totalPrice}
  `;

  alert("Your pizza order has been placed! Total: $" + totalPrice);
}

// Add event listener to the "Order Pizza" button
orderPizzaBtn.addEventListener("click", customizePizzaOrder);
